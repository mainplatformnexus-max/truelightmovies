// server.js
// Railway-ready Relworx backend
// Start command: node server.js

const http = require("http");
const { URL } = require("url");
const crypto = require("crypto");

const PORT = Number(process.env.PORT || 3000);

const RELWORX_BASE_URL =
  process.env.RELWORX_BASE_URL || "https://payments.relworx.com/api";
const RELWORX_API_KEY = process.env.RELWORX_API_KEY || "";
const RELWORX_ACCOUNT_NO = process.env.RELWORX_ACCOUNT_NO || "";
const RELWORX_DEFAULT_CURRENCY = process.env.RELWORX_DEFAULT_CURRENCY || "UGX";

function corsHeaders() {
  return {
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Methods": "GET,POST,OPTIONS",
    "Access-Control-Allow-Headers": "Content-Type,Authorization",
  };
}

function sendJson(res, status, data) {
  res.writeHead(status, {
    ...corsHeaders(),
    "Content-Type": "application/json; charset=utf-8",
  });
  res.end(JSON.stringify(data, null, 2));
}

function makeReference(prefix = "LIVRA") {
  return `${prefix}-${Date.now()}-${crypto.randomBytes(4).toString("hex")}`;
}

function queryString(url) {
  const qs = url.searchParams.toString();
  return qs ? `?${qs}` : "";
}

function requireEnv() {
  const missing = [];
  if (!RELWORX_API_KEY) missing.push("RELWORX_API_KEY");
  if (!RELWORX_ACCOUNT_NO) missing.push("RELWORX_ACCOUNT_NO");

  if (missing.length) {
    const err = new Error(
      `Missing environment variables: ${missing.join(", ")}`
    );
    err.status = 500;
    throw err;
  }
}

function requireFields(obj, fields) {
  const missing = fields.filter(
    (f) => obj?.[f] === undefined || obj?.[f] === null || obj?.[f] === ""
  );
  if (missing.length) {
    const err = new Error(`Missing required fields: ${missing.join(", ")}`);
    err.status = 400;
    throw err;
  }
}

function readBody(req) {
  return new Promise((resolve, reject) => {
    let data = "";
    req.on("data", (chunk) => {
      data += chunk;
      if (data.length > 2 * 1024 * 1024) {
        const err = new Error("Request body too large");
        err.status = 413;
        reject(err);
        req.destroy();
      }
    });
    req.on("end", () => resolve(data));
    req.on("error", reject);
  });
}

async function readJson(req) {
  const raw = await readBody(req);
  if (!raw) return {};
  try {
    return JSON.parse(raw);
  } catch {
    const err = new Error("Invalid JSON body");
    err.status = 400;
    throw err;
  }
}

async function relworx(path, method = "GET", body = null) {
  requireEnv();

  const response = await fetch(`${RELWORX_BASE_URL}${path}`, {
    method,
    headers: {
      "Content-Type": "application/json",
      Accept: "application/vnd.relworx.v2",
      Authorization: `Bearer ${RELWORX_API_KEY}`,
    },
    body: body ? JSON.stringify(body) : undefined,
  });

  const text = await response.text();
  let data;
  try {
    data = text ? JSON.parse(text) : {};
  } catch {
    data = { raw: text };
  }

  if (!response.ok) {
    const err = new Error("Relworx request failed");
    err.status = response.status;
    err.details = data;
    throw err;
  }

  return data;
}

async function fetchPublicIp() {
  const response = await fetch("https://api.ipify.org?format=json");
  if (!response.ok) {
    const err = new Error("Could not fetch public IP");
    err.status = 502;
    throw err;
  }
  return response.json();
}

const server = http.createServer(async (req, res) => {
  try {
    const url = new URL(req.url, `http://${req.headers.host}`);
    const path = url.pathname;
    const method = String(req.method || "GET").toUpperCase();

    if (method === "OPTIONS") {
      res.writeHead(204, corsHeaders());
      return res.end();
    }

    // Root
    if (method === "GET" && path === "/") {
      return sendJson(res, 200, {
        success: true,
        service: "relworx-railway-backend",
        routes: [
          "GET /",
          "GET /health",
          "GET /ip",
          "POST /api/deposit",
          "POST /api/withdraw",
          "POST /api/validate-phone",
          "GET /api/wallet/balance",
          "GET /api/request-status?internal_reference=...",
          "GET /api/transactions",
          "POST /api/card/payment",
          "GET /api/products",
          "GET /api/products/price-list?code=...",
          "GET /api/products/choice-list?code=...",
          "POST /api/products/validate",
          "POST /api/products/purchase",
        ],
      });
    }

    // Health
    if (method === "GET" && path === "/health") {
      return sendJson(res, 200, {
        success: true,
        service: "relworx-railway-backend",
        hasApiKey: Boolean(RELWORX_API_KEY),
        hasAccountNo: Boolean(RELWORX_ACCOUNT_NO),
        baseUrl: RELWORX_BASE_URL,
        defaultCurrency: RELWORX_DEFAULT_CURRENCY,
      });
    }

    // Public IP
    if (method === "GET" && path === "/ip") {
      const ip = await fetchPublicIp();
      return sendJson(res, 200, { success: true, ...ip });
    }

    // MOBILE MONEY DEPOSIT
    if (method === "POST" && path === "/api/deposit") {
      const body = await readJson(req);
      requireFields(body, ["msisdn", "amount"]);

      const data = await relworx("/mobile-money/request-payment", "POST", {
        account_no: RELWORX_ACCOUNT_NO,
        reference: body.reference || makeReference("LIVRA"),
        msisdn: body.msisdn,
        currency: body.currency || RELWORX_DEFAULT_CURRENCY,
        amount: body.amount,
        description: body.description || "Deposit request",
      });

      return sendJson(res, 200, data);
    }

    // MOBILE MONEY WITHDRAW
    if (method === "POST" && path === "/api/withdraw") {
      const body = await readJson(req);
      requireFields(body, ["msisdn", "amount"]);

      const data = await relworx("/mobile-money/send-payment", "POST", {
        account_no: RELWORX_ACCOUNT_NO,
        reference: body.reference || makeReference("WD"),
        msisdn: body.msisdn,
        currency: body.currency || RELWORX_DEFAULT_CURRENCY,
        amount: body.amount,
        description: body.description || "Withdrawal request",
      });

      return sendJson(res, 200, data);
    }

    // VALIDATE PHONE
    if (method === "POST" && path === "/api/validate-phone") {
      const body = await readJson(req);
      requireFields(body, ["msisdn"]);

      const data = await relworx("/mobile-money/validate", "POST", {
        msisdn: body.msisdn,
      });

      return sendJson(res, 200, data);
    }

    // WALLET BALANCE
    if (method === "GET" && path === "/api/wallet/balance") {
      const currency =
        url.searchParams.get("currency") || RELWORX_DEFAULT_CURRENCY;

      const data = await relworx(
        `/mobile-money/check-wallet-balance?account_no=${encodeURIComponent(
          RELWORX_ACCOUNT_NO
        )}&currency=${encodeURIComponent(currency)}`
      );

      return sendJson(res, 200, data);
    }

    // REQUEST STATUS
    if (method === "GET" && path === "/api/request-status") {
      const internalReference = url.searchParams.get("internal_reference");
      if (!internalReference) {
        return sendJson(res, 400, {
          success: false,
          message: "Missing internal_reference",
        });
      }

      const data = await relworx(
        `/mobile-money/check-request-status?internal_reference=${encodeURIComponent(
          internalReference
        )}&account_no=${encodeURIComponent(RELWORX_ACCOUNT_NO)}`
      );

      return sendJson(res, 200, data);
    }

    // TRANSACTIONS
    if (method === "GET" && path === "/api/transactions") {
      const data = await relworx(
        `/payment-requests/transactions?account_no=${encodeURIComponent(
          RELWORX_ACCOUNT_NO
        )}`
      );

      return sendJson(res, 200, data);
    }

    // CARD / VISA SESSION
    if (method === "POST" && path === "/api/card/payment") {
      const body = await readJson(req);
      requireFields(body, ["amount"]);

      const data = await relworx("/visa/request-session", "POST", {
        account_no: RELWORX_ACCOUNT_NO,
        reference: body.reference || makeReference("CARD"),
        currency: body.currency || RELWORX_DEFAULT_CURRENCY,
        amount: body.amount,
        description: body.description || "Card payment",
      });

      return sendJson(res, 200, data);
    }

    // PRODUCTS
    if (method === "GET" && path === "/api/products") {
      const data = await relworx("/products");
      return sendJson(res, 200, data);
    }

    if (method === "GET" && path === "/api/products/price-list") {
      const code = url.searchParams.get("code");
      if (!code) {
        return sendJson(res, 400, { success: false, message: "Missing code" });
      }

      const data = await relworx(
        `/products/price-list?code=${encodeURIComponent(code)}`
      );
      return sendJson(res, 200, data);
    }

    if (method === "GET" && path === "/api/products/choice-list") {
      const code = url.searchParams.get("code");
      if (!code) {
        return sendJson(res, 400, { success: false, message: "Missing code" });
      }

      const data = await relworx(
        `/products/choice-list?code=${encodeURIComponent(code)}`
      );
      return sendJson(res, 200, data);
    }

    if (method === "POST" && path === "/api/products/validate") {
      const body = await readJson(req);
      requireFields(body, ["msisdn", "amount", "product_code"]);

      const data = await relworx("/products/validate", "POST", {
        account_no: RELWORX_ACCOUNT_NO,
        reference: body.reference || makeReference("VAL"),
        msisdn: body.msisdn,
        amount: body.amount,
        product_code: body.product_code,
        contact_phone: body.contact_phone,
        location_id: body.location_id,
      });

      return sendJson(res, 200, data);
    }

    if (method === "POST" && path === "/api/products/purchase") {
      const body = await readJson(req);
      requireFields(body, ["validation_reference"]);

      const data = await relworx("/products/purchase", "POST", {
        account_no: RELWORX_ACCOUNT_NO,
        validation_reference: body.validation_reference,
      });

      return sendJson(res, 200, data);
    }

    return sendJson(res, 404, { success: false, message: "Not Found" });
  } catch (error) {
    return sendJson(res, error.status || 500, {
      success: false,
      error: error.message,
      details: error.details || null,
    });
  }
});

server.listen(PORT, () => {
  console.log(`Relworx backend running on port ${PORT}`);
});
